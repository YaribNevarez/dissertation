38 class MicroInterpreter {
39  public:
40  // The lifetime of the model, op resolver, tensor arena, error reporter and
41  // profiler must be at least as long as that of the interpreter object, since
42  // the interpreter may need to access them at any time. This means that you
43  // should usually create them with the same scope as each other, for example
44  // having them all allocated on the stack as local variables through a
45  // top-level function. The interpreter doesn't do any deallocation of any of
46  // the pointed-to objects, ownership remains with the caller.
47  MicroInterpreter(const Model* model, const MicroOpResolver& op_resolver,
48  uint8_t* tensor_arena, size_t tensor_arena_size,
49  ErrorReporter* error_reporter,
50  MicroProfiler* profiler = nullptr);

52  // Create an interpreter instance using an existing MicroAllocator instance.
53  // This constructor should be used when creating an allocator that needs to
54  // have allocation handled in more than one interpreter or for recording
55  // allocations inside the interpreter. The lifetime of the allocator must be
56  // as long as that of the interpreter object.
57  MicroInterpreter(const Model* model, const MicroOpResolver& op_resolver,
