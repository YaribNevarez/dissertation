170 template <typename T>
171 TfLiteTensor CreateTensor(const T* data, TfLiteIntArray* dims,
172  const bool is_variable = false) {
173  TfLiteTensor result;
174  result.dims = dims;
175  result.params = {};
176  result.quantization = {kTfLiteNoQuantization, nullptr};
177  result.is_variable = is_variable;
178  result.allocation_type = kTfLiteMemNone;
179  result.type = typeToTfLiteType<T>();
180  // Const cast is used to allow passing in const and non-const arrays within a
181  // single CreateTensor method. A Const array should be used for immutable
182  // input tensors and non-const array should be used for mutable and output
183  // tensors.
184  result.data.data = const_cast<T*>(data);
185  result.quantization = {kTfLiteAffineQuantization, nullptr};
186  result.bytes = ElementCount(*dims) * sizeof(T);
187  return result;
188 }

230 // Derives the quantization scaling factor from a min and max range.
231 template <typename T>
232 inline float ScaleFromMinMax(const float min, const float max) {
233  return (max - min) /
234  static_cast<float>((std::numeric_limits<T>::max() * 1.0) -
235  std::numeric_limits<T>::min());
236 }

238 // Derives the quantization zero point from a min and max range.
239 template <typename T>
240 inline int ZeroPointFromMinMax(const float min, const float max) {
241  return static_cast<int>(std::numeric_limits<T>::min()) +
242  static_cast<int>(-min / ScaleFromMinMax<T>(min, max) + 0.5f);
243 }
