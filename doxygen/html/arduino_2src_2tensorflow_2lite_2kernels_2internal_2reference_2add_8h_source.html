43 // Element-wise add that can often be used for inner loop of broadcast add as
44 // well as the non-broadcast add.
45 
46 // This function is used for 8-bit as well as for 16-bit, but the accumulator
47 // is 32-bit for both cases. The overflow does not happen due to the
48 // choice of the shift (20 or 15, accordingly - see add.cc for more comments).
49 template <typename T>
50 inline void AddElementwise(int size, const ArithmeticParams& params,
51  const T* input1_data, const T* input2_data,
52  T* output_data) {
53  TFLITE_DCHECK_GT(params.input1_offset, -std::numeric_limits<T>::max());
54  TFLITE_DCHECK_GT(params.input2_offset, -std::numeric_limits<T>::max());
55  TFLITE_DCHECK_LT(params.input1_offset, std::numeric_limits<T>::max());
56  TFLITE_DCHECK_LT(params.input2_offset, std::numeric_limits<T>::max());
57 
58  for (int i = 0; i < size; ++i) {
59  const int32_t input1_val = params.input1_offset + input1_data[i];
60  const int32_t input2_val = params.input2_offset + input2_data[i];
61  const int32_t shifted_input1_val = input1_val * (1 << params.left_shift);
62  const int32_t shifted_input2_val = input2_val * (1 << params.left_shift);
63  const int32_t scaled_input1_val =
64  MultiplyByQuantizedMultiplierSmallerThanOneExp(
65  shifted_input1_val, params.input1_multiplier, params.input1_shift);
66  const int32_t scaled_input2_val =
67  MultiplyByQuantizedMultiplierSmallerThanOneExp(
68  shifted_input2_val, params.input2_multiplier, params.input2_shift);
69  const int32_t raw_sum = scaled_input1_val + scaled_input2_val;
70  const int32_t raw_output =
71  MultiplyByQuantizedMultiplierSmallerThanOneExp(
72  raw_sum, params.output_multiplier, params.output_shift) +
73  params.output_offset;
74  const int32_t clamped_output =
75  std::min(params.quantized_activation_max,
76  std::max(params.quantized_activation_min, raw_output));
77  output_data[i] = static_cast<T>(clamped_output);
78  }
79 }

84 inline void AddScalarBroadcast(int size, const ArithmeticParams& params,
85  uint8_t input1_data, const uint8_t* input2_data,
86  uint8_t* output_data) {
87  TFLITE_DCHECK_GT(params.input1_offset, -256);
88  TFLITE_DCHECK_GT(params.input2_offset, -256);
89  TFLITE_DCHECK_LT(params.input1_offset, 256);
90  TFLITE_DCHECK_LT(params.input2_offset, 256);
91 
92  const int32_t input1_val = params.input1_offset + input1_data;
93  const int32_t shifted_input1_val = input1_val * (1 << params.left_shift);
94  const int32_t scaled_input1_val =
95  MultiplyByQuantizedMultiplierSmallerThanOneExp(
96  shifted_input1_val, params.input1_multiplier, params.input1_shift);
97  for (int i = 0; i < size; ++i) {
98  const int32_t input2_val = params.input2_offset + input2_data[i];
99  const int32_t shifted_input2_val = input2_val * (1 << params.left_shift);
100  const int32_t scaled_input2_val =
101  MultiplyByQuantizedMultiplierSmallerThanOneExp(
102  shifted_input2_val, params.input2_multiplier, params.input2_shift);
103  const int32_t raw_sum = scaled_input1_val + scaled_input2_val;
104  const int32_t raw_output =
105  MultiplyByQuantizedMultiplierSmallerThanOneExp(
106  raw_sum, params.output_multiplier, params.output_shift) +
107  params.output_offset;
108  const int32_t clamped_output =
109  std::min(params.quantized_activation_max,
110  std::max(params.quantized_activation_min, raw_output));
111  output_data[i] = static_cast<uint8_t>(clamped_output);
112  }

152 inline void Add(const ArithmeticParams& params,
153  const RuntimeShape& input1_shape, const int16_t* input1_data,
154  const RuntimeShape& input2_shape, const int16_t* input2_data,
155  const RuntimeShape& output_shape, int16_t* output_data,
156  bool pot_scale = true) {
157  if (!pot_scale) {
158  AddGeneralParamScale(params, input1_shape, input1_data, input2_shape,
159  input2_data, output_shape, output_data);
160  return;
161  }
162 
163  TFLITE_DCHECK_LE(params.quantized_activation_min,
164  params.quantized_activation_max);
165 
166  const int input1_shift = params.input1_shift;
167  const int flat_size =
168  MatchingElementsSize(input1_shape, input2_shape, output_shape);
169  const int16_t output_activation_min = params.quantized_activation_min;
170  const int16_t output_activation_max = params.quantized_activation_max;
171 
172  TFLITE_DCHECK(input1_shift == 0 || params.input2_shift == 0);
173  TFLITE_DCHECK_LE(input1_shift, 0);
174  TFLITE_DCHECK_LE(params.input2_shift, 0);
175  const int16_t* not_shift_input =
176  input1_shift == 0 ? input1_data : input2_data;
177  const int16_t* shift_input = input1_shift == 0 ? input2_data : input1_data;
178  const int input_right_shift =
179  input1_shift == 0 ? -params.input2_shift : -input1_shift;
180 
181  for (int i = 0; i < flat_size; i++) {
182  // F0 uses 0 integer bits, range [-1, 1].
183  using F0 = gemmlowp::FixedPoint<std::int16_t, 0>;
184 
185  F0 input_ready_scaled = F0::FromRaw(not_shift_input[i]);
186  F0 scaled_input = F0::FromRaw(
187  gemmlowp::RoundingDivideByPOT(shift_input[i], input_right_shift));
188  F0 result = gemmlowp::SaturatingAdd(scaled_input, input_ready_scaled);
189  const int16_t raw_output = result.raw();
190  const int16_t clamped_output = std::min(
191  output_activation_max, std::max(output_activation_min, raw_output));
192  output_data[i] = clamped_output;
193  }
194 }

196 template <typename T>
197 inline typename std::enable_if<!is_small_integer<T>::value, void>::type
198 BroadcastAdd4DSlow(const ArithmeticParams& params,
199  const RuntimeShape& input1_shape, const T* input1_data,
200  const RuntimeShape& input2_shape, const T* input2_data,
201  const RuntimeShape& output_shape, T* output_data) {
202  NdArrayDesc<4> desc1;
203  NdArrayDesc<4> desc2;
204  NdArrayDescsForElementwiseBroadcast(input1_shape, input2_shape, &desc1,
205  &desc2);
206  const RuntimeShape extended_output_shape =
207  RuntimeShape::ExtendedShape(4, output_shape);
208 
209  T activation_min, activation_max;
210  GetActivationParams(params, &activation_min, &activation_max);
211 
212  // In Tensorflow, the dimensions are canonically named (batch_number, row,
213  // col, channel), with extents (batches, height, width, depth), with the
214  // trailing dimension changing most rapidly (channels has the smallest stride,
215  // typically 1 element).
216  //
217  // In generated C code, we store arrays with the dimensions reversed. The
218  // first dimension has smallest stride.
219  //
220  // We name our variables by their Tensorflow convention, but generate C code
221  // nesting loops such that the innermost loop has the smallest stride for the
222  // best cache behavior.
223  for (int b = 0; b < extended_output_shape.Dims(0); ++b) {
224  for (int y = 0; y < extended_output_shape.Dims(1); ++y) {
225  for (int x = 0; x < extended_output_shape.Dims(2); ++x) {
226  for (int c = 0; c < extended_output_shape.Dims(3); ++c) {
227  output_data[Offset(extended_output_shape, b, y, x, c)] =
228  ActivationFunctionWithMinMax<T>(
229  input1_data[SubscriptToIndex(desc1, b, y, x, c)] +
230  input2_data[SubscriptToIndex(desc2, b, y, x, c)],
231  activation_min, activation_max);
232  }
233  }
234  }
235  }

303 inline void BroadcastAddFivefold(const ArithmeticParams& unswitched_params,
304  const RuntimeShape& unswitched_input1_shape,
305  const uint8_t* unswitched_input1_data,
306  const RuntimeShape& unswitched_input2_shape,
307  const uint8_t* unswitched_input2_data,
308  const RuntimeShape& output_shape,
309  uint8_t* output_data) {
310  ArithmeticParams switched_params = unswitched_params;
311  switched_params.input1_offset = unswitched_params.input2_offset;
312  switched_params.input1_multiplier = unswitched_params.input2_multiplier;
313  switched_params.input1_shift = unswitched_params.input2_shift;
314  switched_params.input2_offset = unswitched_params.input1_offset;
315  switched_params.input2_multiplier = unswitched_params.input1_multiplier;
316  switched_params.input2_shift = unswitched_params.input1_shift;
317 
318  const bool use_unswitched =
319  unswitched_params.broadcast_category ==
320  tflite::BroadcastableOpCategory::kFirstInputBroadcastsFast;
321 
322  const ArithmeticParams& params =
323  use_unswitched ? unswitched_params : switched_params;
324  const uint8_t* input1_data =
325  use_unswitched ? unswitched_input1_data : unswitched_input2_data;
326  const uint8_t* input2_data =
327  use_unswitched ? unswitched_input2_data : unswitched_input1_data;
328 
329  // Fivefold nested loops. The second input resets its position for each
330  // iteration of the second loop. The first input resets its position at the
331  // beginning of the fourth loop. The innermost loop is an elementwise add of
332  // sections of the arrays.
333  uint8_t* output_data_ptr = output_data;
334  const uint8_t* input1_data_ptr = input1_data;
335  const uint8_t* input2_data_reset = input2_data;
336  // In the fivefold pattern, y0, y2 and y4 are not broadcast, and so shared
337  // between input shapes. y3 for input 1 is always broadcast, and so the
338  // dimension there is 1, whereas optionally y1 might be broadcast for input 2.
339  // Put another way,
340  // input1.shape.FlatSize = y0 * y1 * y2 * y4,
341  // input2.shape.FlatSize = y0 * y2 * y3 * y4.
342  int y0 = params.broadcast_shape[0];
343  int y1 = params.broadcast_shape[1];
344  int y2 = params.broadcast_shape[2];
345  int y3 = params.broadcast_shape[3];
346  int y4 = params.broadcast_shape[4];
347  if (y4 > 1) {
348  // General fivefold pattern, with y4 > 1 so there is a non-broadcast inner
349  // dimension.
350  for (int i0 = 0; i0 < y0; ++i0) {
351  const uint8_t* input2_data_ptr;
352  for (int i1 = 0; i1 < y1; ++i1) {
353  input2_data_ptr = input2_data_reset;
354  for (int i2 = 0; i2 < y2; ++i2) {
355  for (int i3 = 0; i3 < y3; ++i3) {
356  AddElementwise(y4, params, input1_data_ptr, input2_data_ptr,
357  output_data_ptr);
358  input2_data_ptr += y4;
359  output_data_ptr += y4;
360  }
361  // We have broadcast y4 of input1 data y3 times, and now move on.
362  input1_data_ptr += y4;
363  }
364  }
365  // We have broadcast y2*y3*y4 of input2 data y1 times, and now move on.
366  input2_data_reset = input2_data_ptr;
367  }
368  } else {
369  // Special case of y4 == 1, in which the innermost loop is a single element
370  // and can be combined with the next (y3) as an inner broadcast.
371  //
372  // Note that this handles the case of pure scalar broadcast when
373  // y0 == y1 == y2 == 1. With low overhead it handles cases such as scalar
374  // broadcast with batch (as y2 > 1).
375  //
376  // NOTE The process is the same as the above general case except simplified
377  // for y4 == 1 and the loop over y3 is contained within the
378  // AddScalarBroadcast function.
379  for (int i0 = 0; i0 < y0; ++i0) {
380  const uint8_t* input2_data_ptr;
381  for (int i1 = 0; i1 < y1; ++i1) {
382  input2_data_ptr = input2_data_reset;
383  for (int i2 = 0; i2 < y2; ++i2) {
384  AddScalarBroadcast(y3, params, *input1_data_ptr, input2_data_ptr,
385  output_data_ptr);
386  input2_data_ptr += y3;
387  output_data_ptr += y3;
388  input1_data_ptr += 1;
389  }
390  }
391  input2_data_reset = input2_data_ptr;
392  }
393  }
