115 // Integer arithmetic left-shift, equivalent to multiplying with a power of two.
116 // Negative values are OK. In case of overflow, no Undefined
117 // Behavior, but the results are implementation-defined (in practice,
118 // they currently are saturated, but we make no commitment to that). The idea
119 // is that the caller will want to implement the overflowing cases with
120 // saturation with compare-and-mask, so we don't care about the results
121 // in the overflow case, we just want to avoid undefined behavior.
122 //
123 // tIntegerType may be int32 or any narrower signed type.
124 template <typename tIntegerType>
125 tIntegerType ShiftLeft(tIntegerType a, int offset) {
126  const std::int64_t wide_a = static_cast<std::int64_t>(a);
127  const std::int64_t wide_shifted = wide_a * (1 << offset);
128  const auto min = std::numeric_limits<tIntegerType>::min();
129  const auto max = std::numeric_limits<tIntegerType>::max();
130  return wide_shifted < min
131  ? min
132  : wide_shifted > max ? max
133  : static_cast<tIntegerType>(wide_shifted);
134 }

144 // Each bit of the result is set to the corresponding bit of either then_val or
145 // else_val depending on whether the corresponding bit of if_mask is set.

147 template <typename tIntegerType>

291 // Returns the integer that represents the product of two fixed-point
292 // numbers, interpreting all integers as fixed-point values in the
293 // interval [-1, 1), rounding to the nearest value, and saturating
294 // -1 * -1 to the maximum value (since 1 is not in the half-open
295 // interval [-1, 1)).
296 //
297 // [The explanation below specializes to std::int32_t for example purpose.]
298 //
299 // The mapping between IntegerType and the interval [-1, 1) is unique and
300 // implied by IntegerType, which is assumed to be signed. For example,
301 // for IntegerType==std::int32_t, the mapping is
302 // real_value = integer_value / 2^31.
303 // So in this case, and leaving aside rounding and saturating, this
304 // function computes ((a / 2^31) * (b / 2^31)) * 2^31, which simplifies to
305 // (a * b) / 2^31.
306 //
307 // The 'doubling' part in the name of this function comes from the fact that
308 // this operation is very close to a "multiply-high" operation, keeping only
309 // the top half bits, except that that would be effectively computing
310 // (a * b) / 2^32,
311 // so here we are computing 2x that, since
312 // 1/2^31 = 2 * 1/2^32.
313 // The idea is to use all of the available 32 bits in the destination int32
314 // value.
315 //
316 // [End of the explanation specializing to int32.]
317 //
318 // This is equivalent to the VQRDMULH instruction in ARM NEON.
319 template <typename IntegerType>
320 IntegerType SaturatingRoundingDoublingHighMul(IntegerType a, IntegerType b) {
321  static_assert(std::is_same<IntegerType, void>::value, "unimplemented");
322  (void)b;
323  return a;
324 }
325 
326 // This function implements the same computation as the ARMv7 NEON VQRDMULH
327 // instruction.
328 template <>
329 inline std::int32_t SaturatingRoundingDoublingHighMul(std::int32_t a,
330  std::int32_t b) {
331  bool overflow = a == b && a == std::numeric_limits<std::int32_t>::min();
332  std::int64_t a_64(a);
333  std::int64_t b_64(b);
334  std::int64_t ab_64 = a_64 * b_64;
335  std::int32_t nudge = ab_64 >= 0 ? (1 << 30) : (1 - (1 << 30));
336  std::int32_t ab_x2_high32 =
337  static_cast<std::int32_t>((ab_64 + nudge) / (1ll << 31));
338  return overflow ? std::numeric_limits<std::int32_t>::max() : ab_x2_high32;
339 }
340 
341 template <>
342 inline std::int16_t SaturatingRoundingDoublingHighMul(std::int16_t a,
343  std::int16_t b) {
344  bool overflow = a == b && a == std::numeric_limits<std::int16_t>::min();
345  std::int32_t a_32(a);
346  std::int32_t b_32(b);
347  std::int32_t ab_32 = a_32 * b_32;
348  std::int16_t nudge = ab_32 >= 0 ? (1 << 14) : (1 - (1 << 14));
349  std::int16_t ab_x2_high16 =
350  static_cast<std::int16_t>((ab_32 + nudge) / (1 << 15));
351  return overflow ? std::numeric_limits<std::int16_t>::max() : ab_x2_high16;
352 }

354 // Correctly-rounded-to-nearest division by a power-of-two.
355 // Also known as a rounding arithmetic right shift.
356 template <typename IntegerType>
357 inline IntegerType RoundingDivideByPOT(IntegerType x, int exponent) {
358  assert(exponent >= 0);
359  assert(exponent <= 31);
360  const IntegerType mask = Dup<IntegerType>((1ll << exponent) - 1);
361  const IntegerType zero = Dup<IntegerType>(0);
362  const IntegerType one = Dup<IntegerType>(1);
363  const IntegerType remainder = BitAnd(x, mask);
364  const IntegerType threshold =
365  Add(ShiftRight(mask, 1), BitAnd(MaskIfLessThan(x, zero), one));
366  return Add(ShiftRight(x, exponent),
367  BitAnd(MaskIfGreaterThan(remainder, threshold), one));
368 }

370 // Returns the product of a run-time integer value by a compile-time power
371 // of two, with either a positive exponent (equivalent to an arithmetic
372 // left shift, saturating) or a negative exponent (equivalent to an arithmetic
373 // right shift, rounding to nearest).
374 template <int Exponent, typename IntegerType,
375  int ExponentSign = (Exponent > 0 ? 1 : Exponent < 0 ? -1 : 0)>
376 struct ImplSaturatingRoundingMultiplyByPOT {};
377 
378 template <int Exponent, typename IntegerType>
379 struct ImplSaturatingRoundingMultiplyByPOT<Exponent, IntegerType, 0> {
380  static IntegerType eval(IntegerType x) { return x; }
381 };
382 
383 template <int Exponent, typename IntegerType>
384 struct ImplSaturatingRoundingMultiplyByPOT<Exponent, IntegerType, 1> {
385  static IntegerType eval(IntegerType x) {
386  using ScalarIntegerType =
387  typename FixedPointRawTypeTraits<IntegerType>::ScalarRawType;
388  const IntegerType min =
389  Dup<IntegerType>(std::numeric_limits<ScalarIntegerType>::min());
390  const IntegerType max =
391  Dup<IntegerType>(std::numeric_limits<ScalarIntegerType>::max());
392  const int ScalarIntegerTypeBits = 8 * sizeof(ScalarIntegerType);
393 
394  const std::int32_t threshold =
395  ((1 << (ScalarIntegerTypeBits - 1 - Exponent)) - 1);
396  const IntegerType positive_mask =
397  MaskIfGreaterThan(x, Dup<IntegerType>(threshold));
398  const IntegerType negative_mask =
399  MaskIfLessThan(x, Dup<IntegerType>(-threshold));
400 
401  IntegerType result = ShiftLeft(x, Exponent);
402  result = SelectUsingMask(positive_mask, max, result);
403  result = SelectUsingMask(negative_mask, min, result);
404  return result;
405  }
406 };
407 
408 template <int Exponent, typename IntegerType>
409 struct ImplSaturatingRoundingMultiplyByPOT<Exponent, IntegerType, -1> {
410  static IntegerType eval(IntegerType x) {
411  return RoundingDivideByPOT<IntegerType>(x, -Exponent);
412  }
413 };
414 
415 template <int Exponent, typename IntegerType>
416 IntegerType SaturatingRoundingMultiplyByPOT(IntegerType x) {
417  return ImplSaturatingRoundingMultiplyByPOT<Exponent, IntegerType>::eval(x);
418 }

422 // A FixedPoint object represents a fixed-point value stored in the underlying
423 // integer type tRawType, if tRawType is a plain scalar integer type.
424 // Alternatively, tRawType may be a SIMD type (e.g. NEON int32x4_t) in which
425 // case a FixedPoint object represents a corresponding SIMD vector of fixed
426 // point values.
427 //
428 // tIntegerBits describes the range of the fixed-point format: if
429 // tIntegerBits == m then the range of representable values is the half-open
430 // interval [-2^m; 2^m) where the open boundary on the right side means that
431 // 2^m is not representable (how close the maximum representable value is to
432 // it, depends on bit-depth of tRawType).
433 //
434 // In "Q format notation",
435 // https://en.wikipedia.org/wiki/Q_(number_format)
436 // we are describing the format
437 // Qm.n
438 // where
439 // m = tIntegerBits
440 // and
441 // n = NumberOfBits(tRawType) - (m + 1)
442 // Note that the (m + 1) in the above line is because we adopt the convention
443 // that we count the integer bits exclusively of the sign bit; so (m + 1) is
444 // the total number of integer bits inclusive of the sign bit.
445 //
446 // Accordingly, the number of integral representable values in our range
447 // [-2^m ; 2^m)
448 // is equal to 2^(m+1).
449 template <typename tRawType, int tIntegerBits>
450 class FixedPoint {
451  public:
452  typedef tRawType RawType;
453 
454  typedef FixedPointRawTypeTraits<RawType> RawTypeTraits;
455  typedef typename RawTypeTraits::ScalarRawType ScalarRawType;
456 
457  static constexpr int kTotalBits = 8 * sizeof(ScalarRawType);
458  static constexpr int kIntegerBits = tIntegerBits;
459  static constexpr int kFractionalBits = kTotalBits - 1 - kIntegerBits;
460  static_assert(kIntegerBits >= 0 && kIntegerBits < kTotalBits,
461  "bad IntegerBits");
462 
463  typedef FixedPoint<ScalarRawType, kIntegerBits> ScalarFixedPointType;
464 
465  static const ScalarRawType ScalarRawMin() {
466  return std::numeric_limits<ScalarRawType>::min();
467  }
468 
469  static const ScalarRawType ScalarRawMax() {
470  return std::numeric_limits<ScalarRawType>::max();
471  }
472 
473  static const ScalarRawType RawMin() {
474  return VectorFromScalar(ScalarRawMin());
475  }
476 
477  static const ScalarRawType RawMax() {
478  return VectorFromScalar(ScalarRawMax());
479  }
480 
481  static FixedPoint FromRaw(RawType x) {
482  FixedPoint retval;
483  retval.raw() = x;
484  return retval;
485  }
486 
487  static FixedPoint FromScalarRaw(ScalarRawType x) {
488  FixedPoint retval;
489  retval.raw() = Dup<RawType>(x);
490  return retval;
491  }
492 
493  static FixedPoint FromScalarFixedPoint(ScalarFixedPointType x) {
494  return FromScalarRaw(x.raw());
495  }
496 
497  template <int Exponent>
498  static FixedPoint ConstantPOT() {
499  static constexpr int kOffset = kFractionalBits + Exponent;
500  static_assert(
501  kOffset < 31,
502  "Constant not exactly representable in this fixed-point format");
503  return FromScalarRaw(ScalarRawType(1) << kOffset);
504  }
505 
506  static FixedPoint Zero() { return FromScalarRaw(0); }
507 
508  static FixedPoint One() {
509  return FromScalarRaw(
510  kIntegerBits == 0
511  ? ScalarRawMax()
512  : (ScalarRawType(1) << (kIntegerBits == 0 ? 0 : kFractionalBits)));
513  }
514 
515  static FixedPoint FromDouble(double x) {
516  const double min_bound = static_cast<double>(ScalarRawMin());
517  const double max_bound = static_cast<double>(ScalarRawMax());
518  return FromScalarRaw(static_cast<ScalarRawType>(std::min(
519  std::max(round(x * static_cast<double>(1ll << kFractionalBits)),
520  min_bound),
521  max_bound)));
522  }
523 
524  RawType raw() const { return i_; }
525  RawType& raw() { return i_; }
526 
527  private:
528  RawType i_;
529 };

534 // A FixedPoint multiplication is just a
535 // SaturatingRoundingDoublingHighMul operation on the underlying
536 // raw integer values. The IntegerBits simply add up, as is obvious
537 // from the fact that the range is [-2^IntegerBits, 2^IntegerBits).
538 template <typename tRawType, int tIntegerBits_a, int tIntegerBits_b>
539 FixedPoint<tRawType, tIntegerBits_a + tIntegerBits_b> operator*(
540  FixedPoint<tRawType, tIntegerBits_a> a,
541  FixedPoint<tRawType, tIntegerBits_b> b) {
542  FixedPoint<tRawType, tIntegerBits_a + tIntegerBits_b> c;
543  c.raw() = SaturatingRoundingDoublingHighMul(a.raw(), b.raw());
544  return c;
545 }

676 // CheckedFixedPointConstant allows to specify fixed-point constants
677 // initialized as real numbers, in a way that does not compile floating-point
678 // arithmetic in production code, yet still checks agreement with the
679 // floating-point expressions when asserts are enabled.
680 //
681 // The raw integer value provided is always a int32, encoding a 32-bit
682 // fixed-point value, regardless of the actual Scalar type. This allows
683 // writing generic code that applies just as well to the 32-bit and 16-bit
684 // cases. In the 16-bit case, the raw integer value is internally
685 // rounding-shifted by 16 bits to the right.
686 template <typename FixedPointType>
687 inline typename FixedPointType::ScalarRawType RescaleConstantInitializer(
688  std::int32_t int32_value) {
689  typedef typename FixedPointType::ScalarRawType ScalarRawType;
690  static constexpr int ScalarTypeBits = 8 * sizeof(ScalarRawType);
691  return static_cast<ScalarRawType>(
692  RoundingDivideByPOT<std::int32_t>(int32_value, 32 - ScalarTypeBits));
693 }

719 // Returns exp(x) for x in [-1/4, 0).
720 template <typename tRawType>
721 FixedPoint<tRawType, 0> exp_on_interval_between_negative_one_quarter_and_0_excl(
722  FixedPoint<tRawType, 0> a) {
723  typedef FixedPoint<tRawType, 0> F;
724  const F constant_term =
725  GEMMLOWP_CHECKED_FIXEDPOINT_CONSTANT(F, 1895147668, std::exp(-1.0 / 8.0));
726  const F constant_1_over_3 =
727  GEMMLOWP_CHECKED_FIXEDPOINT_CONSTANT(F, 715827883, 1.0 / 3.0);
728  // We're evaluating a Taylor expansion around -1/8, so we do the change of
729  // variable: x = a + 1/8.
730  // In fixed-point with 0 integer bits, 1/8 is represented by 1 << 28.
731  F x = a + F::template ConstantPOT<-3>();
732  F x2 = x * x;
733  F x3 = x2 * x;
734  F x4 = x2 * x2;
735  F x4_over_4 = SaturatingRoundingMultiplyByPOT<-2>(x4);
736  F x4_over_24_plus_x3_over_6_plus_x2_over_2 =
737  SaturatingRoundingMultiplyByPOT<-1>(
738  ((x4_over_4 + x3) * constant_1_over_3) + x2);
739  return AddSaturatingIf16Bit(
740  constant_term,
741  constant_term * (x + x4_over_24_plus_x3_over_6_plus_x2_over_2));
742 }

744 // Returns exp(x) for x < 0.
745 template <typename tRawType, int tIntegerBits>
746 FixedPoint<tRawType, 0> exp_on_negative_values(
747  FixedPoint<tRawType, tIntegerBits> a) {
748  typedef FixedPoint<tRawType, tIntegerBits> InputF;
749  typedef FixedPoint<tRawType, 0> ResultF;
750  static constexpr int kFractionalBits = InputF::kFractionalBits;
751  static constexpr int kIntegerBits = InputF::kIntegerBits;
752  const InputF kOneQuarter = InputF::template ConstantPOT<-2>();
753  InputF mask = kOneQuarter - InputF::FromScalarRaw(1);
754  InputF a_mod_quarter_minus_one_quarter = (a & mask) - kOneQuarter;
755  ResultF result = exp_on_interval_between_negative_one_quarter_and_0_excl(
756  Rescale<0>(a_mod_quarter_minus_one_quarter));
757  tRawType remainder = (a_mod_quarter_minus_one_quarter - a).raw();
758 
759 #define GEMMLOWP_EXP_BARREL_SHIFTER(Exponent, FixedPointMultiplier) \
760  if (kIntegerBits > Exponent) { \
761  const ResultF kMultiplier = GEMMLOWP_CHECKED_FIXEDPOINT_CONSTANT( \
762  ResultF, FixedPointMultiplier, std::exp(-std::pow(2.0, Exponent))); \
763  static constexpr int kShiftAmount = \
764  kIntegerBits > Exponent ? kFractionalBits + Exponent : 0; \
765  result = SelectUsingMask( \
766  MaskIfNonZero(BitAnd(remainder, Dup<tRawType>(1 << kShiftAmount))), \
767  result * kMultiplier, result); \
768  }
769 
770  GEMMLOWP_EXP_BARREL_SHIFTER(-2, 1672461947);
771  GEMMLOWP_EXP_BARREL_SHIFTER(-1, 1302514674);
772  GEMMLOWP_EXP_BARREL_SHIFTER(+0, 790015084);
773  GEMMLOWP_EXP_BARREL_SHIFTER(+1, 290630308);
774  GEMMLOWP_EXP_BARREL_SHIFTER(+2, 39332535);
775  GEMMLOWP_EXP_BARREL_SHIFTER(+3, 720401);
776  GEMMLOWP_EXP_BARREL_SHIFTER(+4, 242);
777 
778 #undef GEMMLOWP_EXP_BARREL_SHIFTER
779 
780  static constexpr int clampB = kIntegerBits > 5 ? 36 - kIntegerBits : 0;
781  if (kIntegerBits > 5) {
782  const InputF clamp =
783  GEMMLOWP_CHECKED_FIXEDPOINT_CONSTANT(InputF, -(1 << clampB), -32.0);
784  result = SelectUsingMask(MaskIfLessThan(a, clamp), ResultF::Zero(), result);
785  }
786 
787  result = SelectUsingMask(MaskIfZero(a), ResultF::One(), result);
788  return result;
789 }

793 // Returns (1 - x) / (1 + x) for x in (0, 1).
794 template <typename tRawType>
795 FixedPoint<tRawType, 0> one_minus_x_over_one_plus_x_for_x_in_0_1(
796  FixedPoint<tRawType, 0> a) {
797  typedef FixedPoint<tRawType, 0> F0;
798  typedef FixedPoint<tRawType, 2> F2;
799  F0 half_denominator = RoundingHalfSum(a, F0::One());
800  // Newton-Raphson division
801  // https://en.wikipedia.org/wiki/Division_algorithm#Newton.E2.80.93Raphson_division
802  // Refer to that page for the logic behind the 48/17 and 32/17 constants.
803  const F2 constant_48_over_17 =
804  GEMMLOWP_CHECKED_FIXEDPOINT_CONSTANT(F2, 1515870810, 48.0 / 17.0);
805  const F2 constant_neg_32_over_17 =
806  GEMMLOWP_CHECKED_FIXEDPOINT_CONSTANT(F2, -1010580540, -32.0 / 17.0);
807  F2 x = constant_48_over_17 + half_denominator * constant_neg_32_over_17;
808  for (int i = 0; i < 3; i++) {
809  F2 half_denominator_times_x = half_denominator * x;
810  F2 one_minus_half_denominator_times_x =
811  F2::One() - half_denominator_times_x;
812  x = x + Rescale<2>(x * one_minus_half_denominator_times_x);
813  }
814  return Rescale<0>(x - F2::One());
815 }
