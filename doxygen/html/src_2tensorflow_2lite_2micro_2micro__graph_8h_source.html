25 // Abstracts the details of interacting with the tflite::Model.
26 //
27 // Provides methods to access, initialize, prepare, invoke and free any
28 // subgraph in the tflite::Graph.
29 class MicroGraph {
30  public:
31  // The lifetime of the context, model, and allocator must be at least as long
32  // as that of the graph object, since the this class may need to access them
33  // at any time.
34  MicroGraph(TfLiteContext* context, const Model* model,
35  MicroAllocator* allocator);
36  virtual ~MicroGraph();
37 
38  // Sets up builtin data and calls TfLiteRegistration->Init for every operator
39  // in every subgraph in the model.
40  virtual TfLiteStatus InitSubgraphs();
41 
42  // Calls TfLiteRegistration->Prepare for every operator in every subgraph in
43  // the model.
44  virtual TfLiteStatus PrepareSubgraphs();
45 
46  // Calls TfLiteRegistration->Free for every operator in every subgraph in the
47  // model.
48  virtual TfLiteStatus FreeSubgraphs();
49 
50  // Calls TfLiteRegistration->Invoke for every operator in a single subgraph in
51  // the model.
52  virtual TfLiteStatus InvokeSubgraph(int subgraph_idx);
53 
54  // Zeros out all variable tensors in all subgraphs in the model.
55  virtual TfLiteStatus ResetVariableTensors();
56 
57  // Number of tensor inputs to a specified subgraph in the model.
58  virtual size_t NumSubgraphInputs(int subgraph_idx);
59 
60  // Get the specified input tensor of a specified subgraph in the model.
61  virtual TfLiteEvalTensor* GetSubgraphInput(int subgraph_idx, int input_idx);
62 
63  // Number of tensor outputs to a specified subgraph in the model.
64  virtual size_t NumSubgraphOutputs(int subgraph_idx);
65 
66  // Get the specified output tensor of a specified subgraph in the model.
67  virtual TfLiteEvalTensor* GetSubgraphOutput(int subgraph_idx, int output_idx);
68 
69  // Number of subgraphs in the model.
70  virtual int NumSubgraphs();
71 
72  // Hook to pass in subgraph allocations tracked within the interpreter,
73  // allowing MicroGraph to init / prepare / invoke subgraphs in the model.
74  void SetSubgraphAllocations(SubgraphAllocations* subgraph_allocations);
75 
76  // Get the current subgraph index. Within an on operator, this is guaranteed
77  // to be the subgraph of that operator.
78  int GetCurrentSubgraphIndex() { return current_subgraph_index_; }
79 
80  // Gets the list of alloctions for each subgraph. This is the source of truth
81  // for all per-subgraph allocation data.
82  SubgraphAllocations* GetAllocations() { return subgraph_allocations_; }
83 
84  void AllocateEventLogger (void * parent, int subgraph_idx);
85 
86  void DisposeEventLogger(void);
87 
88  private:
89  TfLiteContext* context_;
90  const Model* model_;
91  MicroAllocator* allocator_;
92  SubgraphAllocations* subgraph_allocations_ = nullptr;
93  int current_subgraph_index_;
94  const flatbuffers::Vector<flatbuffers::Offset<SubGraph>>* subgraphs_;
95 
96  void ** event_array_ = nullptr;
97  size_t event_array_len_ = 0;
98 
99  TF_LITE_REMOVE_VIRTUAL_DELETE
100 };
